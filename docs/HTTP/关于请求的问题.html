    # 请求传参转码

    const url = 'http://www.baidu.com?id=abc&name=张三'
下载文件： window.open(url, '_self)
在IE浏览器中，get请求，中文参数不会被转码，（谷歌，火狐浏览器会自动转码utf-8）,
下载文件时，这种情况下，浏览器无法解析中文，IE会报错400，

解决方式：前端需要经过两次转码：
const url = `http://www.baidu.com?id=abc&name=${encodeURI(encodeURI(url))}`
或者：window.open(encodeURIComponent(encodeURIComponent(url)), '_self)

    # ajax 请求步骤（可同步可异步）

    // 1、创建 XMLHttpRequest 异步对象
let xhr;
if (window.XMLHttpRequest) {
  xhr = new XMLHttpRequest()
} else {
  // IE6
  xhr = new ActiveXObject(("Microsoft.XMLHTTP")
}

// 2、设置回调函数
xhr.onreadystatechange = function() {
  if (xhr.readyState === 4) {
    if (xhr.status === 200) {
      // 服务器响应成功
      const res = xhr.responseText
    }
  }
}

// readyState	存有 XMLHttpRequest 的状态。从 0 到 4 发生变化。
// -	0: 请求未初始化
// -	1: 服务器连接已建立
// -	2: 请求已接收
// -	3: 请求处理中
// -	4: 请求已完成，且响应已就绪

// 3、使用open方法，与服务器建立连接
// get 方式
xhr.open('get', 'http://www.baidu.com/getList', true)

// post 方式
xhr.open('post', 'http://www.baidu.com/post', true)
// post请求下需要配置请求头信息
// xmlHttp.setRequestHeader("Content-Type", "application/x-www-form-urlencoded");

// 设置响应数据类型，得到响应后后自动解析。
// 目前支持的类型有string、arraybuffer、blob、document、json、text、ms-stream。
xhr.responseType('json')

// 4、发送数据
// get 请求
xhr.send()

// post请求下，要传递的参数放这
xhr.send("userName=tset&str1=str1&str2=str2");

// 中断请求
xhr.abort()

    # fetch(异步)

    1.fetch是基于promise实现的，也可以结合async/await

2.fetch请求默认是不带cookie的，需要设置: fetch(URL, {credentials: 'include'})
Credentials有三种参数：same-origin，include，*

3.服务器返回400 500 状态码时并不会reject，只有网络出错导致请求不能完成时，fetch才会被reject

4.所有版本的 IE 均不支持原生 Fetch， Edge可以支持

5.fetch是window的一个方法

    function postRequest(url) {
  //将"key1=value1&key2=valu2" 形式封装整FromData形式
  let formData = new FormData();
  formData.append("username", "hello");
  formData.append("password", "1111aaaa");

  var opts = {
    method: "POST",
    body: formData,
    headers: new Headers({
      Accept: "application/json",
      "Content-Type": "application/json",
    }),
  };

  fetch(url, opts)
    .then((response) => {
      // 你可以在这个时候将Promise对象转换成json对象:response.json()
      // 可以格式化的数据类型有： json  text   blob[ 二进制 ], formData(), arrayBuffer()
      // 转换成json对象后return，给下一步的.then处理
      return response.text;

      // 获取头部信息
      // console.log(response.status);
      // console.log(response.statusText);
      // console.log(response.headers.get('Content-Type'));
      // console.log(response.headers.get('Date'));
    })
    .then((responseText) => {
      alert(responseText);
    })
    .catch((error) => {
      alert(error);
    });
}
