    YJcoders 5/8/2021 DAG

    # 一、功能需求分析

    1、实现 ETL 任务调度功能， 2、组件列表动态可控，支持拖拽组件到画布 3、组件之间可连线，并可删除连线 4、组件可编辑设置属性，可删除组件 5、组件的连接点数量，是否允许连接可控 6、画布支持放大缩小 7、节点展示信息支持定制 8、每个节点的数据结构设计 9、可以对整个 dag 进行相关设置

    # 二、ux、ui 设计

    1、使用常用的布局，左侧组件树，可搜索，可拖拽 2、右侧画布区，节点拖拽移动，编辑删除等其他操作 3、右侧顶部设置 toolbar 可对整个 dag 进行保存，属性设置等操作

    # 三、功能实现

    1、左侧组件树 2、右侧画布区 3、右侧 toolbar

    # 左侧组件树

    使用 ant-tree 树组件 + Dnd 拖拽组件

    • 树节点按下后，调用 Dnd 的 start 方法开始拖拽
    • 拖拽到画布区，放置后设置连接点，设置节点相关属性
    <template>
  <a-directory-tree default-expand-all>
    <!-- 定义分组 -->
    <a-tree-node
      v-for="(value, name, index) in componentComputed"
      :key="index"
      :title="groupName[name]"
      :icon="getIcon"
    >
      <!-- 定义组件 -->
      <a-tree-node
        v-for="item in value"
        :key="item.configPageCode"
        :title="compName[item.configPageCode]"
        is-leaf
        data-type="rect"
        :icon="getIcon"
        @mousedown.native="startDrag"
      />
    </a-tree-node>
  </a-directory-tree>
</template>

<script>
export default {
  methods: {
    startDrag(e) {
      const text = e.target.innerText;
      const { compName } = this;
      const id = this.dataSource.filter(
        (item) => compName[item.configPageCode] === text
      )[0].configPageCode;
      // 创建节点
      const node = CreateNode(`dropdown-component-${id}`, this);
      this.$graph.resizeNode(node, text);
      // 鼠标按下，树节点，开始拖拽，拖拽到画布，放置节点
      this.$dnd.start(node, e);
    },
  },
};
</script>

    拖拽组件定义

    import { Addon } from "@antv/x6";
const { Dnd } = Addon;

const dnd = new Dnd({
  target: graph, // 目标画布
  scaled: false,
  animation: true,
  getDropNode: this.getDropNode, // 定义拖拽结束时，所放置节点的信息
});

getDropNode(node) {
  const cloneNode = node.clone();
  // 设置连接点
  this.resetPorts(cloneNode, maxInputPoint, "in");
  this.resetPorts(cloneNode, maxOutputPoint, "out");
  // 设置节点信息
  this.addBizData(cloneNode);
  return cloneNode;
},

addBizData(node) {
  const componentInfo = this.getComponentInfo(node);
  const ins = node.getPortsByGroup("in").map(item => item.id);
  const outs = node.getPortsByGroup("out").map(item => item.id);
  const stages = {
    dagId: this.dagId,
    stageId: node.id,
    checkPassed: true,
    remark: node.remark || "",
    remarkVisible: node.remarkVisible || true,
    ...componentInfo,
    // eslint-disable-next-line camelcase
    input_point_ids: [...ins],
    // eslint-disable-next-line camelcase
    out_point_ids: [...outs],
    inputLanes: [],
    outputLanes: [...outs],
  };
  node.setData({ stages });
},

    # 右侧画布区

    使用 Graph 初始化画布

    • 初始化时，定义画布网格，拖动，连接线等样式
    • 校验连接线是否可连接，连接是否生效
    import { Graph } from "@antv/x6";

// 初始化画布
this.$graph = new Graph({
  container: document.getElementById("container"),
  grid: {
    visible: true,
    type: "mesh",
  },
  panning: true,
  history: true,
  snapline: {
    enabled: true,
    sharp: true,
  },
  mousewheel: {
    enabled: true,
    modifiers: ["ctrl", "meta"],
    minScale: "0.3", // 最小缩放比例
  },
  connecting: {
    anchor: "center",
    connectionPoint: "anchor",
    allowBlank: false,
    allowMulti: false,
    allowLoop: false,
    allowNode: false,
    allowEdge: false,
    allowPort: true,
    highlight: true,
    snap: true,
    // 设置连接线属性
    createEdge() {
      return new Shape.Edge({
        defaultLabel: {
          markup: Markup.getForeignObjectMarkup(),
          attrs: {
            fo: {
              width: 120,
              height: 30,
              x: 60,
              y: -15,
            },
          },
        },
        attrs: {
          line: {
            stroke: variables.labelColor,
            strokeWidth: 1,
            targetMarker: {
              name: "block",
              size: 8,
            },
          },
        },
        router: {
          name: "manhattan",
          args: {
            startDirections: ["right"],
            endDirections: ["left"],
          },
        },
        connector: {
          name: "rounded",
          args: {
            radius: 5,
          },
        },
        zIndex: -1,
      });
    },
    // 设置是否允许作为起点连线
    validateMagnet({ magnet }) {
      return magnet.getAttribute("port-group") !== "in";
    },
    // 校验连线是否生效，不生效会被去除
    validateEdge({ edge, type, previous }) {
      const sourceNode = edge.getSourceCell();
      const targetNode = edge.getTargetCell();
      // 如果目标是质检组件，必须输入是数据源
      if (
        sourceNode.getData().stages.configPageCode != "TABLE_INPUT" &&
        targetNode.getData().stages.configPageCode == "DATA_CHECK"
      ) {
        UiUtils.confirm({
          title: "前置组件必须为表输入",
          content: "",
          class: "confirm-modal",
          okText: "确定",
          cancelText: "取消",
          onOk: () => {},
        });
        return false;
      }
      return true;
    },
    // 连线时，鼠标松开后，判断是否有效，无效不会连接
    validateConnection({ sourceView, targetView, sourceMagnet, targetMagnet }) {
      if (!sourceMagnet || sourceMagnet.getAttribute("port-group") === "in") {
        return false;
      }
      if (!targetMagnet || targetMagnet.getAttribute("port-group") !== "in") {
        return false;
      }
      const portId = targetMagnet.getAttribute("port");
      const targetNode = targetView.cell;
      const port = targetNode.getPort(portId);
      if (port && port.connected) {
        return false;
      }
      return true;
    },
  },
});

    initEvent(graph) {
  // 鼠标移入，移出时 高亮设置
  graph.on("cell:mouseenter", ({ cell }) => {
    if (cell.isEdge()) {
      this.setCellAttrs(cell, true);
    } else {
      this.nodeSetCellAttrs(cell, true);
    }
  });
  graph.on("cell:mouseleave", ({ cell }) => {
    if (cell.isEdge()) {
      cell.removeTools();
      this.setCellAttrs(cell, false);
    } else {
      this.nodeSetCellAttrs(cell, false);
    }
  });
  // 连线之后的，起止节点的属性设置
  graph.on("edge:connected", ({ edge, currentCell, currentMagnet }) => {

  });
  // 删除连线之后的，起止节点的属性设置
  graph.on("edge:removed", ({ edge, currentCell, currentMagnet }) => {

  });
}
