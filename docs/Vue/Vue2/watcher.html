    # watcher 分类

    • 计算属性 watcher (computed)
    • 监听 watcher（watch）
    • 渲染 watcher (render)

    执行顺序： 计算属性 watcher -> 监听 watcher -> 渲染 watcher

    # 源码位置

    // new vue -> init -> initState(initComputed -> initWatch) -> $mount(mountComponent -> render watcher)
export function initState(vm: Component) {
  vm._watchers = [];
  const opts = vm.$options;
  if (opts.props) initProps(vm, opts.props);
  if (opts.methods) initMethods(vm, opts.methods);
  if (opts.data) {
    initData(vm);
  } else {
    observe((vm._data = {}), true /* asRootData */);
  }
  if (opts.computed) initComputed(vm, opts.computed);
  if (opts.watch && opts.watch !== nativeWatch) {
    initWatch(vm, opts.watch);
  }
}

    # 2、Watcher 源码

    export default class Watcher {
  vm: Component;
  expression: string; //
  cb: Function;
  id: number;
  deep: boolean;
  user: boolean; // watch Watcher
  lazy: boolean; // computed Watcher
  sync: boolean;
  dirty: boolean; // computed Watcher 为true时会重新计算取值，否则取缓存值
  active: boolean;
  deps: Array<Dep>; // 存储依赖Dep
  newDeps: Array<Dep>; // 存储依赖Dep 区别是此次更新后，收集到的下一轮更新相关的依赖
  depIds: SimpleSet; // 存储deps对应的dep的id所组成的一个Set
  newDepIds: SimpleSet;
  before: ?Function;
  getter: Function; // 获取监听的value的函数
  value: any; // 监听的对象的值

  constructor(
    vm: Component,
    expOrFn: string | Function,
    cb: Function,
    options?: ?Object,
    isRenderWatcher?: boolean // 是否是渲染watcher，只有mount函数里面过来的才是true
  ) {
    this.vm = vm;
    if (isRenderWatcher) {
      vm._watcher = this;
    }
    vm._watchers.push(this);
    // options
    if (options) {
      this.deep = !!options.deep;
      this.user = !!options.user;
      this.lazy = !!options.lazy;
      this.sync = !!options.sync;
      this.before = options.before;
    } else {
      this.deep = this.user = this.lazy = this.sync = false;
    }
    this.cb = cb;
    this.id = ++uid; // uid for batching
    this.active = true;
    this.dirty = this.lazy; // for lazy watchers
    this.deps = [];
    this.newDeps = [];
    this.depIds = new Set();
    this.newDepIds = new Set();
    this.expression =
      process.env.NODE_ENV !== "production" ? expOrFn.toString() : "";
    // parse expression for getter
    if (typeof expOrFn === "function") {
      this.getter = expOrFn;
    } else {
      this.getter = parsePath(expOrFn);
      if (!this.getter) {
        this.getter = noop;
      }
    }
    // value取值，区分 computed 和 render， watch
    // computed watcher初始化的时候，不会直接取值，值为undefined
    this.value = this.lazy ? undefined : this.get();
  }

  // render watch 对应的watcher 才会调用 get
  get() {
    // 1、取出当前的watcher
    pushTarget(this);
    let value;
    const vm = this.vm;
    try {
      // 2、取值，触发defineReactive的get 进行依赖收集
      value = this.getter.call(vm, vm);
    } catch (e) {
    } finally {
      // 深度监听
      if (this.deep) {
        traverse(value);
      }
      // 3、运行完之后清空当前的watcher
      popTarget();
      this.cleanupDeps();
    }
    return value;
  }

  //  watcher添加dep，然后里面dep存放对应的watcher
  addDep(dep: Dep) {
    const id = dep.id;
    if (!this.newDepIds.has(id)) {
      this.newDepIds.add(id);
      this.newDeps.push(dep);
      if (!this.depIds.has(id)) {
        dep.addSub(this);
      }
    }
  }

  // 当前watcher的依赖收集完毕，需要将newDepIds和newDeps的值分别赋给depIds和deps，
  // 并且将newDepIds和newDeps的值清空，等待下一次页面更新时，重新收集依赖
  cleanupDeps() {
    let i = this.deps.length;
    while (i--) {
      const dep = this.deps[i];
      if (!this.newDepIds.has(dep.id)) {
        dep.removeSub(this);
      }
    }
    let tmp = this.depIds;
    this.depIds = this.newDepIds;
    this.newDepIds = tmp;
    this.newDepIds.clear();
    tmp = this.deps;
    this.deps = this.newDeps;
    this.newDeps = tmp;
    this.newDeps.length = 0;
  }

  // 数据更新，watcher触发视图更新
  update() {
    if (this.lazy) {
      // computed watcher
      // watcher监听的值变化了，但并不会立即执行，也不会加入到异步更新队列
      // 等到渲染读取的时，再去取值
      this.dirty = true;
    } else if (this.sync) {
      // watch watcher
      // 立即执行watcher实例的run方法
      this.run();
    } else {
      // render watcher
      // 将该watcher实例加入到异步更新队列中
      queueWatcher(this);
    }
  }

  run() {
    if (this.active) {
      // 获取watcher 修改后 value
      const value = this.get();
      if (value !== this.value || isObject(value) || this.deep) {
        // set new value
        const oldValue = this.value;
        this.value = value;
        if (this.user) {
          try {
            // watch watcher
            this.cb.call(this.vm, value, oldValue);
          } catch (e) {
            handleError(
              e,
              this.vm,
              `callback for watcher "${this.expression}"`
            );
          }
        } else {
          this.cb.call(this.vm, value, oldValue);
        }
      }
    }
  }

  // computed watcher
  // createComputedGetter方法中为计算属性定义了取值器(computedGetter)
  // 当访问计算属性时，如果computed watcher实例的dirty属性为true，将会调用evaluate方法，重新计算计算属性的值
  // 取值之后，设置dirty为false
  evaluate() {
    // 取值，触发defineReactive数据劫持，进行依赖收集
    this.value = this.get();
    this.dirty = false;
  }

  // 依赖收集
  depend() {
    let i = this.deps.length;
    while (i--) {
      this.deps[i].depend();
    }
  }

  // 移除所有监听
  teardown() {
    if (this.active) {
      if (!this.vm._isBeingDestroyed) {
        remove(this.vm._watchers, this);
      }
      let i = this.deps.length;
      while (i--) {
        this.deps[i].removeSub(this);
      }
      this.active = false;
    }
  }
}

    # 3、observer 源码

    export class Observer {
  value: any;
  dep: Dep;
  vmCount: number; // number of vms that have this object as root $data

  constructor(value: any) {
    this.value = value;
    this.dep = new Dep();
    this.vmCount = 0;
    def(value, "__ob__", this);
    // 数组单独处理
    if (Array.isArray(value)) {
      if (hasProto) {
        protoAugment(value, arrayMethods);
      } else {
        copyAugment(value, arrayMethods, arrayKeys);
      }
      this.observeArray(value);
    } else {
      this.walk(value);
    }
  }

  // 遍历data， 添加数据劫持
  walk(obj: Object) {
    const keys = Object.keys(obj);
    for (let i = 0; i < keys.length; i++) {
      defineReactive(obj, keys[i]);
    }
  }

  //
  observeArray(items: Array<any>) {
    for (let i = 0, l = items.length; i < l; i++) {
      observe(items[i]);
    }
  }
}

// 通过 Object.defineProperty 数据劫持
export function defineReactive(
  obj: Object,
  key: string,
  val: any,
  customSetter?: ?Function,
  shallow?: boolean
) {
  const dep = new Dep();

  let childOb = !shallow && observe(val);
  Object.defineProperty(obj, key, {
    enumerable: true,
    configurable: true,
    get: function reactiveGetter() {
      const value = getter ? getter.call(obj) : val;
      // 当前有且，唯一的 watcher是，为此watcher添加依赖收集
      if (Dep.target) {
        dep.depend();
        if (childOb) {
          childOb.dep.depend();
          if (Array.isArray(value)) {
            dependArray(value);
          }
        }
      }
      return value;
    },
    set: function reactiveSetter(newVal) {
      const value = getter ? getter.call(obj) : val;
      if (newVal === value || (newVal !== newVal && value !== value)) {
        return;
      }
      if (getter && !setter) return;
      if (setter) {
        setter.call(obj, newVal);
      } else {
        val = newVal;
      }
      childOb = !shallow && observe(newVal);
      // 数据更新，触发视图更新
      dep.notify();
    },
  });
}

function protoAugment(target, src: Object) {
  target.__proto__ = src;
}
function copyAugment(target: Object, src: Object, keys: Array<string>) {
  for (let i = 0, l = keys.length; i < l; i++) {
    const key = keys[i];
    def(target, key, src[key]);
  }
}

    # 4、Dep 源码

    export default class Dep {
  static target: ?Watcher;
  id: number;
  subs: Array<Watcher>;

  constructor() {
    this.id = uid++;
    this.subs = [];
  }

  addSub(sub: Watcher) {
    this.subs.push(sub);
  }

  removeSub(sub: Watcher) {
    remove(this.subs, sub);
  }

  depend() {
    if (Dep.target) {
      Dep.target.addDep(this);
    }
  }

  notify() {
    const subs = this.subs.slice();
    if (process.env.NODE_ENV !== "production" && !config.async) {
      subs.sort((a, b) => a.id - b.id);
    }
    for (let i = 0, l = subs.length; i < l; i++) {
      subs[i].update();
    }
  }
}

Dep.target = null;
const targetStack = [];
// 确保Dep.target指向的是当前正在可能对data进行取值操作的watcher，将这个watcher暴露出来，方便被取值的data进行依赖收集
export function pushTarget(target: ?Watcher) {
  // push当前watcher到stack 中，并记录当前watcer
  targetStack.push(target);
  Dep.target = target;
}
export function popTarget() {
  // 运行完之后清空当前的watcher
  targetStack.pop();
  Dep.target = targetStack[targetStack.length - 1];
}
