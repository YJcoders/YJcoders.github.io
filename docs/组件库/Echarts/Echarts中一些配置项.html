    # echarts 在 vue 中使用

    1、安装
npm install echarts --save
2、在main.js引入
import echarts from 'echarts'
3、在main.js全局注册
Vue.prototype.$echarts = echarts
4、页面中使用

const option = {}
this.domCharts.setOption(option)

    # 柱状图缩放 dataZoom, (数据量过大，但是又要全部显示，需要使用 dataZoom,滚动查看)

    // 设置页面需要固定显示的bar 数量,此处设置 20
// xData:  缩放显示的数据
// dataZoomEnd:  数据窗口范围的结束百分比。范围是：0 ~ 100
// 先计算dataZoomEnd
let dataZoomEnd
if (xData.length > 20) {
  // 大于20条,显示20
  dataZoomEnd = (20 / xData.length) * 100
} else {
  // 小于20条全显示
  dataZoomEnd = 100
}

// 然后在option中设置

dataZoom: {
  realtime: true, // 拖动滚动条实时更新图表数据
  height: 12, // 滚动条高度,
  start: 0, // 滚动条开始位置（共100份）
  end: dataZoomEnd, // 结束位置
}

    # x 坐标轴 label 标签显示换行处理（两个字换行）

    xAxis: {
  axisLable: {
    show: true,
    formatter: function(data) {
      const strs = data.split('')
      let str = ''
      strs.forEach((item, index) => {
        const i = index + 1
        str += item
        if (!(i % 2)) str += '\n'
      })
      return str
    }
  }
}

    # 监听鼠标移动（点击）事件，获取鼠标当前位置

    this.chart3.on("click", (params) => {
  // 获取点击的数据
  console.log(params, "999");
});

// 监听鼠标的移动，点击等事件
this.chart3.getZr().on("mousemove", (params) => {
  const pointInpixel = [params.offsetX, params.offsetY];
  if (this.chart3.containPixel("grid", pointInpixel)) {
    // 获取当前鼠标位置
    const xIndex = this.chart3.convertFromPixel({ seriesIndex: 0 }, [
      params.offsetX,
      params.offsetY,
    ])[0];
    if (xIndex > 2) {
      // 设置鼠标样式
      this.chart3.getZr().setCursorStyle("move");
    }
  }
});
// 使用getZr添加图表的整个canvas区域的点击事件，并获取params携带的信息：
this.chart1.getZr().on("click", (params) => {
  // 获取到鼠标点击位置 ：
  const pointInPixel = [params.offsetX, params.offsetY];
  // 使用containPixel API判断点击位置是否在显示图形区域，下面的例子过滤了绘制图形的网格外的点击事件，比如X、Y轴lable、空白位置等的点击事件。
  if (this.chart1.containPixel("grid", pointInPixel)) {
    // 使用API convertFromPixel获取点击位置对应的x轴数据的索引值，详细请查看文档。
    const xIndex = this.chart1.convertFromPixel({ seriesIndex: 0 }, [
      params.offsetX,
      params.offsetY,
    ])[0];
    // 在上一步骤中可以获取到丰富的诸如轴线、索引、ID等信息，可以在自己的事件处理代码中方便的使用
    console.log(xIndex);
  }
});

// 监听多柱状图滚动
this.chart4.on("dataZoom", (params) => {
  const data = this.chart4.getOption().dataZoom[0];
  this.startIndex = data.startValue;
  this.endIndex = data.endValue;
});
