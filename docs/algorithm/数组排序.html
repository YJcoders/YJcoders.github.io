    # 1、冒泡排序

    时间复杂度 O(n^2) 空间复杂度 O(1) 每次内循环，相邻的两个元素对比，会将最大（或最小）元素放置最右 再外循环，最后一个元素就不用再次对比

    function bubbleSort(arr) {
  const len = arr.length;
  for (let i = 0; i < len; i++) {
    for (let j = 0; j < len - i - 1; j++) {
      if (arr[j] > arr[j + 1]) {
        [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];
      }
    }
  }
  return arr;
}

    # 2、选择排序

    时间复杂度 O(n^2) 空间复杂度 O(1) 定义临时索引变量为每次外循环的元素最小（大），内循环如果比外循环元素大（或小），则将内循环索引赋值到临时索引变量 每次内循环完成，对比临时索引变量和外层循环索引的对应值，如果不相等 则调换元素

    function selectSort(arr) {
  const len = arr.length;
  for (let i = 0; i < len; i++) {
    let min = i;
    for (let j = i + 1; j < len; j++) {
      if (arr[j] < arr[min]) {
        min = j;
      }
    }

    if (i != min) [arr[i], arr[min]] = [arr[min], arr[i]];
  }
  return arr;
}

    # 3、插入排序

    时间复杂度 O(n^2) 空间复杂度 O(1) 将数组元素一分为二，初始时第一个元素为一组，剩下元素为一组 遍历剩下的每个元素，作为临时值，和此元素的前一个的元素比较，如果比前面的大（或小），则前面的元素后移，继续和前面的前一个元素比较， 最后将临时值，插入移动后的空位

    function insertSort(arr) {
  let len = arr.length;
  // 注意 起始索引是 1
  for (let i = 1; i < len; i++) {
    let j = i;
    let temp = arr[i]; // 存储剩下元素作为临时值
    // 临时值和前面的元素进行比较
    while (j > 0 && arr[j - 1] < temp) {
      // 满足条件，前面的元素，向后移动
      arr[j] = arr[j - 1];
      // 继续和前面的前一个元素比较
      j--;
    }
    // 最后将临时值，插入到移出的空位
    arr[j] = temp;
  }
  return arr;
}

    # 4、快速排序

    时间复杂度 O(nlogn) 空间复杂度 O(logn) 采用分治法思想 选择一个元素作为基准值，比较其他元素，将元素分为 大于 和 小于 该基准值的 两个数组 然后再递归使用此方法，直到剩下最后一个元素 比较过程： 1、选取第一个元素作为初始值，使用 start 和 end 双指针 标记第一个元素和最后一个元素 2、先从最右开始和基准值比较，找到比基准值大的（或小）填入 start 指向的元素位置（同时 start 指针右移一位），否则 end 指针左移继续比较 3、然后开始从左侧比较，找到比基准值小的（或大）填入 end 指向的元素位置（同时 end 指针左移一位），否则 start 指针右移继续比较 4、第一轮比较结束之后，分为两个数组，然后再递归分别对比两组数据

    function quick(arr) {
  return sort(arr, 0, arr.length - 1);

  function sort(arr, start, end) {
    if (start >= end) return;

    let i = start;
    let j = end;
    const base = arr[start];

    while (i < j) {
      while (i < j) {
        // 1、先从右侧开始比较，找到比基准值大的值，将值填入左指针对应数据
        // 并且左指针右移一位，跳出循环
        if (arr[j] > base) {
          arr[i] = arr[j];
          i++;
          break;
        }
        // 未找到，右指针左移，继续查找
        j--;
      }

      while (i < j) {
        // 2、再从左侧开始查找，找到比基准值小的值，将值填入右指针对应的数据
        // 并且右指针左移一位，跳出循环
        if (arr[i] < base) {
          arr[j] = arr[i];
          j--;
          break;
        }
        // 未找到，左指针右移，继续查找
        i++;
      }

      // 3、继续循环比较直到 i == j
    }
    // 4、将base基准值，填入空位i
    arr[i] = base;
    // 5、继续递归，两组数据，进行排序
    sort(arr, start, i - 1);
    sort(arr, i + 1, end);

    return arr;
  }
}

    # 5、归并排序

    时间复杂度 O(nlogn) 空间复杂度 O(logn) 采用分治思想 先将数组拆分成小数组，再拆分为单个元素，然后使用选择排序的方式，对拆分的小数组进行合并排序

    function mergeSort(arr) {
  return sort(arr, 0, arr.length - 1);
  // sort方法，进行递归
  function sort(arr, left, right) {
    // 当left !== right时，表示还没分拆到最小元素
    if (left < right) {
      // 取中间值，分拆为两个小的数组
      const mid = Math.floor((left + right) / 2);
      const leftArr = sort(arr, left, mid);
      const rightArr = sort(arr, mid + 1, right);
      // 递归合并
      return merge(leftArr, rightArr);
    }
    // left == right, 已经是最小元素，直接返回即可
    return left >= 0 ? [arr[left]] : [];
  }

  // 将左右两个有序数组 合并 排序
  // i 指向 左数组   j 指向右数组
  function merge(leftArr, rightArr) {
    let i = 0;
    let j = 0;
    const result = [];

    // 使用双指针，对两个数组进行扫描
    while (i < leftArr.length && j < rightArr.length) {
      if (leftArr[i] >= rightArr[j]) {
        result.push(leftArr[i]);
        i++;
      } else {
        result.push(rightArr[j]);
        j++;
      }
    }

    // 合并剩下的数据
    while (i < leftArr.length) {
      result.push(leftArr[i++]);
    }
    while (j < rightArr.length) {
      result.push(rightArr[j++]);
    }

    return result;
  }
}

    # 6、希尔排序

    时间复杂度 O(nlog^2n) 空间复杂度 O(1) 是基于插入排序的优化 按增量对数组元素进行分组，然后对分组的元素进行插入排序 随着增量的减小，直至为 1，所有元素又被分为一组，然后再进行少量排序

    function shellSort() {
  // 获取增量
  let gap = Math.floor(arr.length / 2);
  // 增量等于1 时 插入排序
  while (gap >= 1) {
    // 从第gap个元素开始，逐个对其所在组进行直接插入排序
    for (let i = gap; i < arr.length; i++) {
      let j = i;
      let temp = arr[i];
      while (j > gap - 1 && arr[j - gap] < temp) {
        arr[j] = arr[j - gap];
        j -= gap;
      }

      arr[j] = temp;
      console.log(gap, j, temp, arr, "jj");
    }
    // 缩小增量
    gap = Math.floor(gap / 2);
  }
  return arr;
}

    # 7、计数排序

    时间复杂度 O(n + k) 空间复杂度 O(k) 只能用来排序 整数 利用空间换时间 过程： 1、创建一个新数组，长度满足最大值 + 偏移量，值都填充为 0 2、遍历数组，将数值，按(索引 + 偏移量)存入新数组，并标记值为 1，如果有重复数据，标记值+1（相当于已经排序） 3、再遍历新数组，遇到值不为 0 的，直接取 (索引 - 偏移量) 得到的值，放置新的空数组中，即为排序完成

    function countingSort(arr) {
  let maxValue = Number.MIN_VALUE;
  let minValue = Number.MAX_VALUE;
  let offset = 0; // 偏移量，用于处理负数
  const result = [];

  // 取出数组中的最大值, 最小值
  arr.forEach((num) => {
    maxValue = num > maxValue ? num : maxValue;
    minValue = num > minValue ? minValue : num;
  });
  if (minValue < 0) {
    offset = -minValue;
  }

  // 初始化新的数组，数组长度 最大元素 + 偏移量
  const bucket = new Array(maxValue + offset + 1).fill(0);
  // 将数组中的每个值 加上偏移量，放入对应新数组的索引
  // 加偏移量是为了处理 负数
  arr.forEach((num) => {
    bucket[num + offset]++;
  });
  // 读取新数组
  bucket.forEach((store, index) => {
    while (store--) {
      // 对应索引，减去偏移量，即为原始值
      result.push(index - offset);
    }
  });

  return result;
}

    # 8、桶排序

    时间复杂度 O(n) 空间复杂度 O(k) 基于计数排序的优化，分治法 + 空间换时间 将数组分组，然后对子分组进行排序，最后再合并

    function bucketSort(arr, bucketSize = 10) {
  // bucketSize 每个桶可以存放的数字区间(0, 9]

  if (arr.length <= 1) {
    return arr;
  }

  let maxValue = arr[0];
  let minValue = arr[0];
  let result = [];

  // 取出数组的最大值, 最小值
  arr.forEach((num) => {
    maxValue = num > maxValue ? num : maxValue;
    minValue = num > minValue ? minValue : num;
  });

  // 初始化桶的数量
  const bucketCount = Math.floor((maxValue - minValue) / bucketSize) + 1; // 桶的数量
  // 初始化桶的容器
  // 注意这里的js语法，不能直接fill([])，因为生成的二维下标数组，是同一个地址
  const buckets = new Array(bucketCount).fill(0).map(() => []);

  // 将数字按照映射的规则，放入桶中
  arr.forEach((num) => {
    const bucketIndex = Math.floor((num - minValue) / bucketSize);
    buckets[bucketIndex].push(num);
  });

  // 遍历每个桶内存储的数字
  buckets.forEach((store) => {
    // 桶内只有1个数字或者空桶，或者都是重复数字，则直接合并到结果中
    if (store.length <= 1 || bucketSize == 1) {
      result = result.concat(store);
      return;
    }

    // 递归，将桶内的数字，再进行一次划分到不同的桶中
    const subSize = Math.floor(bucketSize / 2); // 减少桶内的数字区间，但必须是最少为1
    const tmp = bucketSort(store, subSize <= 1 ? 1 : subSize);
    result = result.concat(tmp);
  });

  return result;
}
