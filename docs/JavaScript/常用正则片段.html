    # 1、正则片段

    去除字符串中的空格: str.replace(/\s+/g, '')
将连续的空格替换为一个空格: str.replace(/\s+/g, ' ')
只能输入英文（否则替换为空） str.replace(/[^a-zA-Z]/g, '')
只能输入数字（否则替换为空） str.replace(/[^0-9]/g, '')
只能输入英文，数字（否则替换为空） str.replace(/[^\w]/g, '')
只能输入英文，数字、中文（否则替换为空） str.replace(/^[\w\u4e00-\u9fa5]+$/g, '')
只能输入英文，数字，下划线，且只能以英文开头： /^[a-zA-z][a-zA-Z0-9_]*$/
匹配0-9以外的任意字符： /[^0-9]/

不能以下划线开头： /^(?!_)/
不能以数字开头： /^(?![0-9])/
不能以下划线结尾: (?!.*?_$)
不能以下划线开头的数字，英文，下划线： /^(?!_)[a-zA-Z0-9_]*$/

不能以空格开头，不能以空格结尾， 中间可以输入空格：/^(?!\s)[\w\s]{2,50}(?<!\s)$/
不以某字符串开头：?! 否定式向前查询  /^(?!str)/,    /^(?!str)/.test('streee') false
不以某字符串结尾：?<! 否定式向后查询  /(?<!str)$/,    /(?<!str)$/.test('ttttstr') false
肯定式向前查找：(?=str)；肯定式向后查找：(?<=str)

任意字符，长度3-6： /^.{3,6}$/

校验ip: const reg = /^((2[0-4]\d|25[0-5]|[01]?\d\d?)\.){3}(2[0-4]\d|25[0-5]|[01]?\d\d?)$/;
        const flag = reg.test(ip);

校验域名： /^http(s)?:\/\/[\w_-].(.+?)/

在页面的空格 &nbsp; 通过dom.innerText传给后端，后端在脚本编译器shell里面无法识别
需要转义传递：
const sql = this.dom.innerText
const text = decodeURI(encodeURI(sql).replace(/%C2%A0/g, '20%'))


// 校验身份证
// 弱校验，正则
/^[1-8][1-7]\d{4}(?:19|20)\d{2}(?:0[1-9]|1[0-2])(?:0[1-9]|[12]\d|3[01])\d{3}[\dX]$/.test('110101199003078844')
// 强校验 方法
function checkIDCard(idcode) {
  // 加权因子
  var weight_factor = [
  7,
  9,
  10,
  5,
  8,
  4,
  2,
  1,
  6,
  3,
  7,
  9,
  10,
  5,
  8,
  4,
  2
  ];
  // 校验码
  var check_code = [
  "1",
  "0",
  "X",
  "9",
  "8",
  "7",
  "6",
  "5",
  "4",
  "3",
  "2"
  ];

  var code = idcode + "";
  var last = idcode[17]; //最后一位

  var seventeen = code.substring(0, 17);

  // ISO 7064:1983.MOD 11-2
  // 判断最后一位校验码是否正确
  var arr = seventeen.split("");
  var len = arr.length;
  var num = 0;
  for (var i = 0; i < len; i++) {
  num = num + arr[i] * weight_factor[i];
  }

  // 获取余数
  var resisue = num % 11;
  var last_no = check_code[resisue];

  // 格式的正则
  // 正则思路
  /*
  第一位不可能是0
  第二位到第六位可以是0-9
  第七位到第十位是年份，所以七八位为19或者20
  十一位和十二位是月份，这两位是01-12之间的数值
  十三位和十四位是日期，是从01-31之间的数值
  十五，十六，十七都是数字0-9
  十八位可能是数字0-9，也可能是X
  */
  var idcard_patter = /^[1-9][0-9]{5}([1][9][0-9]{2}|[2][0][0|1][0-9])([0][1-9]|[1][0|1|2])([0][1-9]|[1|2][0-9]|[3][0|1])[0-9]{3}([0-9]|[X])$/;

  // 判断格式是否正确
  var format = idcard_patter.test(idcode);

  // 返回验证结果，校验码和格式同时正确才算是合法的身份证号码
  return last === last_no && format ? true : false;
}
  console.log(checkIDCard("420621199210036332"));
