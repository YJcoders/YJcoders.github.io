    # 4、bind/call/apply/ 修改 this 指向,实现对此方法的借用

    1、三者区别与共同点
三个都是用来改变函数this的指向，
第一个参数都是this要指向的对象，也就是想指定的上下文；
区别：
call/apply  会直接调用函数；bind只是返回一个新的函数，不会调用
call和apply传参方式不同
call参数是一个参数列表，
apply参数是一个数组，内部其实也是将数组转换为了 参数列表
call:
const target = {}
fn.call(target, 'arg1', 'arg2')
apply:
const target = {}
fn.apply(target, ['arg1', 'arg2'])

// 例1
let o1 = {
    a: 1
};
let o2 = {
    a: 2
};
let o3 = {
    a: 3
};

function fn(b, c) {
    console.log(this.a);
};

let fn1 = fn.bind(o1);
let fn2 = fn1.bind(o2);
let fn3 = fn2.bind(o3);
fn3() // 1

---
bind多次绑定，函数this还是指向第一次bind的对象。
---

2、常见用法
--- 获取最大值，最小值
const nums = [2,10,100,-20]
// Math.max  Math.min 接收的参数是一个参数列表
// 所以需要使用apply，把数组转换为参数列表
const min = Math.min.apply(null, nums)

--- 合并数组
var arr1 = [1, 2, { id: 1, id1: 2 }, [1, 2]];
var arr2 = ['ds', 1, 9, { name: 'jack' }];
// var arr = arr1.concat(arr2);
Array.prototype.push.apply(arr1,arr2)

--- 判断变量类型
function isArray(obj) {
  return Object.prototype.toString.call(obj) == '[object Array]'
}
isArray([]) //true
Object.prototype.toString.call(new Date)  // [object Date]
Object.prototype.toString.call(new String)  // [object String]

--- 操作类数组
常见类数组（具有length属性，但是没有数组方法）：
arguments, DOM列表，
定义一个类数组：
const arr = {0:"name", 1:12, 2:true, 3:["测试", "test", 99], length:4 };
Array.prototype.slice.call(arr) // 类数组转换为数组
Array.prototype.slice.call(arr, 1, 3) // 截取类数组参数
Array.from(arr)  // es6
[...arguments]  // es6解构

3、模拟实现call
Function.prototype.createCall = function(context) {
  // 防止 Function.prototype.createCall() 直接调用
  if (this === Function.prototype) return undefined
  // 取得传入的对象（执行上下文）,如果为空，默认取Window
  context = context || window;
  // 保证第一个参数是对象,不是就创建一个对象
  if (typeof context !== 'object') {
    context = Object.create(null)
  }
  // 给context添加一个属性，这时的this指向调用call的函数
  // 添加的属性要唯一，不能覆盖调用对象上的属性
  const fn = Symbol()
  context[fn] = this;
  // 通过展开运算符和解构赋值取出context后面的参数
  var args = [...arguments].slice(1);
  // 执行函数
  var result = context[fn](...args);
  // 删除函数
  delete context[fn];
  return result;
};

// 例
function a(age) {
    console.log(this.name, 'name')
    console.log(this.fn, 'fn')
    console.log(age, 'age')
}
const obj = {
    name: 'zs',
    fn: 123
}
function d() {}
a.createCall(obj, 12)
a.createCall('obj')
a.createCall(d)


3、模拟实现apply (和call区别就是处理参数)
Function.prototype.createApply = function(context) {
  // 防止 Function.prototype.createApply() 直接调用
  if (this === Function.prototype) return undefined
  const args = [...arguments].slice(1)

  if (!(args instanceof Array)) throw Error('参数格式错误')

  // 取得传入的对象（执行上下文）,如果为空，默认取Window
  context = context || window;
  // 保证第一个参数是对象,不是就创建一个对象
  if (typeof context !== 'object') {
    context = Object.create(null)
  }
  // 给context添加一个属性，这时的this指向调用call的函数
  // 添加的属性要唯一，不能覆盖调用对象上的属性
  const fn = Symbol()
  context[fn] = this;

  let result;
  if (!args.length) {
    result = context[fn]()
  } else {
    result = context[fn](...args)
  }
  // 删除函数
  delete context[fn]
  return result
};


4、模拟实现bind
Function.prototype.createBind = function(context) {
  if(typeof this !== 'function') {
      throw new TypeError('Error')
    }
    var self = this;
    var args = [...arguments].slice(1)

    var fun = function() {
        // 当作为构造函数的时候，this 指向实例， self指向绑定函数，由fun.prototype = this.prototype, 使 fun.prototype 为 绑定函数的 prototype，此时结果为true，所以this指向实例
        // 作为普通函数时，this 指向window， self 指向绑定函数，此时结果为false，此时this指向绑定的context
        self.apply(this instanceof self ? this : context,args.concat(...arguments))
    }
    // 修改返回函数的 prototype 为绑定函数的 prototype，实例就可以继承函数的原型中的值
    // 使用Object.create避免修改函数的prototype
    fun.prototype = Object.create(this.prototype)
    return fun
}

// 例
var value = 2;
var foo = {
    value: 1
}
function bar(name, age) {
    console.log(name, 'name')
    console.log(this, '注意此处的 this')
    console.log(this.value)
}

bar.prototype.sex = 'boy'
var bindFoo = bar.bind2(foo)
// bindFoo('zs')   //直接调用方式
var obj1 = new bindFoo(23) // 通过new构造函数方式

    # 5、构造函数（new）中的 this

    // 构造函数，函数名首字母要大写
function Foo() {
    this.bar = "name"
}
const instance = new Foo()
console.log(instance.bar) // name

new 操作符 调用构造函数过程中做了什么？
1、创建一个新的对象
2、将构造函数的this指向这个对象
3、执行构造函数中的代码，为这个对象添加属性
4、返回这个对象

// 代码实现new 构造函数
// function create() {
//     // 1、创建一个空对象 obj
//     let obj = {}; // 或 let obj = new Object()
//     // 2、取到第一个参数,此参数为传入的构造函数
//     let Constructor = [].shift.call(arguments);
//     // 3、将该对象 obj 的原型链 __proto__ 指向构造函数的原型 prototype
//     // 使其可以访问构造函数原型中的属   性
//     obj.__proto__ = Constructor.prototype
//     // 4、改变构造函数中的this指向 obj
//     // 此时obj也可以访问构造函数中的属性
//     let result = Constructor.apply(obj, arguments)
//     // 确保返回值是一个对象
//     return typeof result === 'object' ? result : obj
// }
function create() {
    // 1、创建一个空对象 obj
    let obj = {}; // 或 let obj = new Object()
    // 2、取到第一个参数,此参数为传入的构造函数
    let Constructor = [].shift.call(arguments);
    // 3、将该对象 obj 的原型链 __proto__ 指向构造函数的原型 prototype
    // 使其可以访问构造函数原型中的属   性
    obj.__proto__ = Constructor.prototype
    // 4、改变构造函数中的this指向 obj
    // 此时obj也可以访问构造函数中的属性
    let result = Constructor.apply(obj, arguments)
    // 确保返回值是一个对象
    return typeof result === 'object' ? result : obj
}

// 示例
function Person(name) {
    this.name = name
}
Person.prototype.say = function () {
    console.log(this.name)
}

let obj = create(Person, 'zs')
console.log(obj.name) // zs
console.log(obj.__proto__ === Person.prototype) // true

    # 6、箭头函数的 this

    箭头函数使用 this 是根据外层（函数或者全局）上下文来决定
箭头函数无法使用 apply / call / bind 进行绑定 this 值
无法使用 new 实例化对象，因为普通构造函数通过 new 实例化对象时 this 指向实例对象，
而箭头函数没有 this 值，同时 箭头函数也没有 prototype。
// 例1
var obj={
	fn:function(){
		setTimeout(function(){
			console.log(this);
		});
	}
}
obj.fn(); // Window

// 例2
var obj={
	fn:function(){
		setTimeout(() => {
			console.log(this);
		});
	}
}
obj.fn() // obj

箭头函数的this一旦确定指向后，无法再被修改
// 例
function foo() {
    return a => {
        console.log(this.a)
    };
}
const obj1 = {
    a: 2
}
const obj2 = {
    a: 3
}

const bar = foo.call(obj1) // 2
bar.call(obj2) // 2

    # 7、this 指向的优先级

    通过 call、apply、bind、new 对 this 绑定的情况称为显式绑定；根据调用关系确定的 this 指向称为隐式绑定。
显示绑定优先级 >  隐式绑定

// 例1
function foo (a) {
    console.log(this.a)
}

const obj1 = {
    a: 1,
    foo: foo
}

const obj2 = {
    a: 2,
    foo: foo
}

obj1.foo.call(obj2) // 2
obj2.foo.call(obj1) // 1
