# 1、props

父向子传递数据 props 最好使用对象写法，可以设置属性类型和默认值

// 父组件 Parent
<template>
  <div>
    <child  :name="name" />
  </div>
</template>

// 子组件 Child
export default { 
  props: {
    name: {
      type: String,
      default: '',
    }
  }
}

子组件修改 props 传递的数据
方法1：子组件通过 $emit 方法，传递数据，父组件监听该方法，拿到值，在进行修改

// 子组件：
export default { 
  methods: {
    changeName() {
      this.$emit('changeName', '子组件修改值')
    }
  }
}
// 父组件
<template>
  <div>
    <child  :name="name" @changeName="change" />
  </div>
</template>

<script>
export default { 
  methods: {
    change(val) {
      this.name = val
    }
  }
}
</script>

# 2、provide / inject

这对选项需要一起使用，以允许一个祖先组件向其所有子孙后代注入一个依赖
不论组件层次有多深，并在起上下游关系成立的时间里始终生效。
provide 和 inject 是不可响应的(但是：如果传入了一个可监听的对象，那么其对象的属性还是可响应的,对象是因为是引用类型)

// 父组件
export default {
  provide() {
    return {
      name: '父组件定义数据',
      app: this, // 可以将整个vue传递下去
    }
  }
}

// 子组件
export default {
  inject: ['name', 'this'],
  mounted () {
    console.log(this.name);  // 父组件定义数据
  }
}

# 3、ref 和 $parent 、 $children

这三个不适合跨级通信，只有一层父子级之间通信可以

<!-- 父组件 -->
<template>
  <div>
    <child  ref='child' />
  </div>
</template>

<script>
export default { 
  mounted() {
    // 获取子组件实例
    console.log(this.$refs.child)

    // 获取所有子组件
    console.log(this.$children)
  }
}
</script>

<!-- 子组件 -->
<script>
export default { 
  mounted() {
    // 获取父组件实例
    console.log(this.$parent)
  }
}
</script>

# 4、EventBus (事件总线)

所有组件共用事件中心，可以向该中心注册发送事件或接收事件，所以组件都可以上下平行地通知其他组件

// 初始化全局事件总线 bus.js
import Vue from 'vue'
export const EventBus = new Vue()

// 或者全局注册
Vue.prototype.$EventBus = new Vue()

// $emit 发送消息
EventBus.$emit("message", '我是消息');
// $on 监听消息
EventBus.$emit("message", (msg) => {});
// 移除指定事件监听
EventBus.$off('message')
// 移除所有事件监听
EventBus.$off()

A组件发送消息

<!-- A页面 -->
<template>
  <button @click="send()">-</button>
</template>

<script> 
import { EventBus } from "./bus.js";
export default {
  methods: {
    send() {
      EventBus.$emit("message", '我是A组件');
    }
  }
}; 
</script>

<!-- B页面接收消息 -->
<template>
  <p>{{msg}}</p>
</template>

<script> 
import { EventBus } from "./bus.js";
export default {
  data(){
    return {
      msg: ''
    }
  },
  mounted() {
    EventBus.$on("message", (msg) => {
      // A发送来的消息
      this.msg = msg;
    });
  }
};
</script>

# 5、$attrs 和 $listeners

// 父组件，
<child title="标题" content="内容" num="12" @change="change" />

// 子组件可以通过 $attrs拿到这些属性，$listeners拿到方法，前提是这些属性没有在子组件中被 props 接收
mounted() {
  console.log(this.$attrs); // {title: "标题", content: "内容", num: "12"}
  console.log(this.$listeners); // {change: ƒ}
}

// 如果props接收了某个属性，$attrs就会剔除这个属性
props: {
  num: {
    type: String,
    default: ''
  }
},
mounted() {
  console.log(this.$attrs) // {title: "标题", content: "内容"}
},
