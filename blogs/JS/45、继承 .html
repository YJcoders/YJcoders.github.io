# 1、借用构造函数继承

function Person(name) {
  this.like = ['eat', 'say']
  this.name = name
}
Person.prototype.sayName = function() {
  console.log(this.name)
}
function Child(name) {
  Person.call(this, name)
}

const child1 = new Child('zs')
console.log(child1.name)
child1.like.push('smile')
console.log(child1.like)
console.log(child1.sayName) // undefined
const child2 = new Child('ls')
console.log(child2.name)
console.log(child2.like)

---优点
1、避免引用类型的属性被所有实例共享
2、可以在Child中向Person传参
---缺点
1、不能继承父类函数的原型属性和方法
2、所以方法只能写在构造函数内部，导致每次创建实例就会创建一次方法

# 2、原型继承

function Person(name) {
  this.like = ['eat', 'say']
  this.name = name
}
Person.prototype.sayName = function() {
  console.log(this.name)
}
Person.prototype.num = '333'
function Child() {
  // 区别构造函数位置
}

// 区别构造函数位置 --- 原型继承
Child.prototype = new Person()

const child1 = new Child('zs')
console.log(child1)
console.log(child1.num)
child1.sayName()
child1.like.push('smile') // 实例child1修改属性
console.log(child1.like)
const child2 = new Child('ls')
console.log(child2.like) // 实例child2 也受到影响

---优点
1、可以继承父类原型链上的属性和方法
2、
---缺点
1、引用类型的属性共享，一个修改，都会受影响
2、子类不能像父类传参

# 3、组合继承（构造函数 + 原型）

function Person(name) {
  this.fruit = ['apple', 'orange']
  this.name = name
}
Person.prototype.sayName = function() {
  console.log(this.name)
}
Person.prototype.like = ['eat', 'say']
function Child(name, age) {
  // 继承父类的 实例属性
  Person.call(this, name)
  this.age = age
}

// 修改子类原型，继承父类原型方法
Child.prototype = new Person()
// 再将原型属性的 constructor属性指回 Child
Child.prototype.constructor = Child

const child1 = new Child('zs', 12)
// child1.num.a = 7
child1.sayName()
child1.fruit.push('bb')
child1.like.push('smile')
// console.log(child1.num)
console.log(child1)
const child2 = new Child('ls', 34)
console.log(child2)

---优点
1、可以继承父类属性和方法，以及原型链上的属性和方法
2、父类属性相互独立
3、子类可向父类传参
---缺点
1、原型链上的引用类型的属性共享，一个修改，都会受影响
2、实例化时，父类被构造了两次；call一次，new一次

# 4、寄生式

function createObj (obj) {
  // 基于传入的基础对象，创建一个新的对象
  var clone = Object.create(obj);
  // 增强对象，为新建对象添加属性，方法
  clone.sayName = function () {
      console.log('hi');
  }
  // 最后返回这个对象
  return clone;
}

var person = {
  name: 'zs'
}
var one = createObj(person)
console.log(one)

1、
---缺点
1、添加的属性，方法不能复用

# 5、寄生式组合继承 --- 目前最优的继承方式

// <js高级程序设计>
所谓寄生式组合继承，就是通过借用构造函数来继承属性，通过原型链来继承原型方法。
不必为了指定子类型的原型而调用超类型的构造函数，利用寄生式继承来继承超类型的原型，然后再将结果指定给子类型的原型

function Parent (name, ss) {
    this.name = name;
    this.ss = ss
    this.fruit = ['apple', 'orange']
}

Parent.prototype.sayName = function () {
    console.log(this.name)
}

Parent.prototype.like = ['eat', 'say']
function Child (name, age, ss) {
    Parent.call(this, name, ss);
    this.age = age

}

// 关键的三步,
// 通过寄生式继承来继承Parent原型链
Child.prototype = Object.create(Parent.prototype);
// 修改原型属性constructor的指向
Child.prototype.constructor = Child

// 构造函数继承属性
const child1 = new Child('zs', 12)
// child1.num.a = 7
child1.sayName()
child1.fruit.push('bb')
child1.like.push('smile')
// console.log(child1.num)
console.log(child1)
const child2 = new Child('ls', 34, 'ss')
console.log(child2)

console.log(child1 instanceof Child)  // true
console.log(child1 instanceof Parent) // true

# 6、ES6的 Class extends --- 结果等同于 寄生式组合继承

区别：
寄生式组合继承：是先创建子类实例对象this,然后再增强属性
Class extends: 先将父类实例对象，方法挂在this上（所以必须先调用super方法继承this），然后再用子类的构造函数修改this

其实ES6的继承是一种语法糖，本质仍然是构造函数+原型链的组合式继承。

// 例
class A {
    constructor(name, age) {
        this.name = name;
        this.age = age;
        this.arr = [1, 2]
    }
    getName() {
        return this.name;
    }
}

class B extends A {
    constructor(name, age) {
        super();
        this.job = "work";
        this.array = [4, 5]
    }
    getJob() {
        return this.job;
    }
    getNameAndJob() {
      console.log(this)
        return this.getName() + this.job;
    }
}

var b = new B('zs', 18)
// console.log(b.name)
// console.log(b.getName())
// console.log(b.getNameAndJob())
b.arr.push(3)
b.array.push(7)
console.log(b.arr) // [1, 2, 3]
console.log(b.array) // [4 ,5, 7]
var c = new B('ls', 30)
console.log(c.arr) // [1, 2]
console.log(c.array) // [4, 5]

1、extends 关键字
通过extends关键字，继承了A类的所有属性和方法。A类中的所有方法默认是添加到B的原型上，所以extends继承的实质仍然是原型链。

2、super 关键字
super，既可以当作函数使用，也可以当作对象使用。当作函数使用时，super代表父类的构造函数，并在子类中执行Parent.apply(this)，从而将父类实例对象的属性和方法，添加到子类的this上面。
以下三点需要特别注意：
2.1
子类必须在constructor方法中调用super方法，如果子类没有定义constructor方法，constructor方法以及其内部的super方法会被默认添加。
2.2
在子类的constructor方法中，只有调用super之后，才可以使用this关键字，否则会报错
2.3
super()方法， 注意是方法，只能用在子类的constructor方法之中，用在其他地方就会报错。
其他地方可以使用他的对象调用方式：super.getName()

3、static 静态方法关键字
在父类方法前面加一个static关键字，该方法不会被实例继承，但是会被子类继承

class A {
    static say() {
        console.log("hello");
    }
}

class B extends A {}

console.log(B.say()); // hello
var b = new B()
console.log(b.say()); // 报错，b.say is not a function
