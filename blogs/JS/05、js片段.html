# 1、JSON.stringify()在转换数据，会丢失值为undefined，NaN和函数的属性

处理方式：使用JSON.stringify的第二个参数

语法：JSON.stringify(value[, replacer[, space]])
value: 必需， 要转换的值（通常为对象或数组）

replacer: 可选，用于转换结果的函数或数组。
如果 replacer 为函数，则 JSON.stringify 将调用该函数，并传入每个成员的键和值。使用返回值而不是原始值。
如果 replacer 是一个数组，则仅转换该数组中具有键值的成员。成员的转换顺序与键在数组中的顺序一样。当 value 参数也为数组时，将忽略 replacer 数组。

space: 可选，文本添加缩进、空格和换行符，如果 space 是一个数字，则返回值文本在每个级别缩进指定数目的空格，如果space大于 10，则文本缩进10个空格。space 也可以使用非数字，如：\t

var ff = {
  aa: 'aa'
}
var obj = {
  a: 4,
  b: 'bbb',
  c: null,
  d: undefined,
  e: function () {
    return 'www'
  },
  f: ff,
  g: true
}

function replacer (key, value) {
  if (value === undefined) return 'undefined'
  if (typeof(value) === 'function') {
    return Function.prototype.toString.call(value)
  }
  return value
}

let obj1 = JSON.stringify(obj)
let obj2 = JSON.stringify(obj, replacer)
console.log(JSON.parse(obj1), 'obj1');
console.log(JSON.parse(obj2), 'obj2');

# 3、获取地址栏参数

function getUrlParams(url) {
  if (!url) { return };
  var result = ''
  var index = window.location.href.indexOf('?')
  var searchArr = window.location.href.slice(index + 1).split('&')
  searchArr.map(function (item) {
    var arr = item.split('=')
    if (arr[0] === url) {
      result = decodeURIComponent(arr[1])
    }
  })
  return result
}

# 4、截取地址栏token, 替换地址， 并存储在本地

function getURLToken(name) {
  if (!name) return ''
  const href = decodeURIComponent(window.location.href)
  const index = href.lastIndexOf('?') + 1
  const arr = href.slice(index).split('&')
  arr.map(item => {
    const temp = item.split('=')
    if (temp[0] === name) {
      const result = temp[1]
      // 处理有些情况下返回地址栏参数添加的有 '?' 可能会影响跳转到其他页面的地址栏参数 
      window.location.href = arr[0].length && arr.length <=1 ? href.replace(`?${temp[0]}=${temp[1]}`, '')
        : href.replace(`${temp[0]}=${temp[1]}`, '')

      window.localStorage.setItem('token', result)
    }
  })
}

# 8、encodeURI和encodeURIComponent

URI: 通用资源标识符;
encodeURI()和encodeURIComponent()方法都可以对URI进行编码

但是编码有所不同：
encodeURI()方法不会对下列字符编码：a-z、A-Z、0-9、~!@#$&*()=:/,;?+'
encodeURIComponent()方法不会对下列字符编码：a-z、A-Z、0-9、-_.~!*()'

对应解码：
encodeURI()           => decodeURI()
encodeURIComponent()  => decodeURIComponent()

# 9、Math函数使用

# 11、校验是否是标准的JSON格式

function isJSON(str) {
  if (typeof str === 'string) {
    try {
      let obj = JSON.parse(str)
      if (typeof obj === 'object' && obj) {
        return true
      } else {
        return false
      }
    } catch (e) {
      return false
    }
  }
}

# 12、判断浏览器类型

function checkBrowser () {
  var ua = navigator.userAgent.toLocaleLowerCase()
  var browserType = null
  if (ua.match(/msie/) != null || ua.match(/trident/) != null) {
    browserType = 'IE'
    const browserVersion = ua.match(/msie ([\d.]+)/) != null ? ua.match(/msie ([\d.]+)/)[1] : ua.match(/rv:([\d.]+)/)[1]
  } else if (ua.match(/firefox/) != null) {
    browserType = 'Firefox'
  } else if (ua.match(/ubrowser/) != null) {
    browserType = 'UC'
  } else if (ua.match(/opera/) != null) {
    browserType = 'Opera'
  } else if (ua.match(/bidubrowser/) != null) {
    browserType = 'Baidu'
  } else if (ua.match(/metasr/) != null) {
    browserType = 'Sougou'
  } else if (ua.match(/tencenttraveler/) != null || ua.match(/qqbrowse/) != null) {
    browserType = 'QQ'
  } else if (ua.match(/maxthon/) != null) {
    browserType = 'Aoyou'
  } else if (ua.match(/chrome/) != null) {
    var is360 = _mime('type', 'application/vnd.chromium.remoting-viewer')
    if (is360) {
      browserType = '360'
    } else {
      browserType = 'Google'
    }
  } else if (ua.match(/safari/) != null) {
    browserType = 'Safari'
  }
  function _mime (option, value) {
    var mimeTypes = navigator.mimeTypes
    for (var mt in mimeTypes) {
      if (mimeTypes[mt][option] === value) {
        return true
      }
    }
    return false
  }
  return browserType
}

// 2、使用其他方式
function formatCash(str) {
  return str.split('').reverse().reduce((prev, next, index) => {
    return ((index % 3) ? next : (next + ',')) + prev
  })
}
console.log(format) // 1,234,567,890

toFullscreen() {
  this.isFullScreen = !this.isFullScreen;
  const app = document.getElementById('app');
  if (this.isFullScreen) {
    this.openFullscreen(app);
  } else {
    this.exitFullScreen();

// 打开全屏方法
openFullscreen(element) {
  console.log(element.webkitRequestFullscreen);
  if (element.requestFullscreen) {
    element.requestFullscreen();
  } else if (element.mozRequestFullScreen) {
    element.mozRequestFullScreen();
  } else if (element.msRequestFullscreen) {
    element.msRequestFullscreen();
  } else if (element.webkitRequestFullscreen) {
    element.webkitRequestFullScreen();

  } else if (document.mozCancelFullScreen) {

  } else if (document.webkitCancelFullScreen) {
    document.webkitCancelFullScreen();
  } else if (document.webkitExitFullscreen) {
    document.webkitExitFullscreen();
  }
}

# 15、点击按钮复制文本到粘贴板

handleCopy() {
  if (window.clipboardData) {
    // IE
    // 清除粘贴板
    window.clipboardData.clearData();
    //将需要复制的内容复制到操作系统粘贴板
    window.clipboardData.setData("Text", "要复制的内容");
  } else {
    var input = document.createElement("input");

    document.body.appendChild(input);
    input.select();
    var res = document.execCommand("copy");
    document.body.removeChild(input);
    return res;
  }
}

# 16、下载文本为txt或其他格式

handleDownload() {
  const res = "223";
  const name = "example.txt";
  if (window.navigator.msSaveBlob) {
    const blob = new Blob([res], {
      type: "text/json"
    });
    window.navigator.msSaveBlob(blob, name);
  } else {
    const blob = new Blob([res], {
      type: "text/json"
    });
    const url = window.URL.createObjectURL(blob);
    const link = document.createElement("a");
    link.href = url;
    link.setAttribute("download", name);
    link.click();
  }
}
